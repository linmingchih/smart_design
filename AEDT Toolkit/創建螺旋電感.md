創建螺旋電感
---
用錄製方式任意在AEDT Layout任意畫一條多邊線，將錄製好程式碼丟給AI，讓他產生螺旋電感及UI。30分鐘搞定。

```python
# -*- coding: utf-8 -*-
import sys
import os
import math
import json

import clr
clr.AddReference("System.Windows.Forms")
clr.AddReference("System.Drawing")
from System.Windows.Forms import (Application, Form, Label, TextBox, Button, 
                                  DialogResult, MessageBox, ComboBox, ComboBoxStyle,
                                  FormBorderStyle, FormStartPosition, FlatStyle)
from System.Drawing import Point, Size, Color, Font, FontStyle

try:
    import ScriptEnv
    ScriptEnv.Initialize("Ansoft.ElectronicsDesktop")
    oDesktop.RestoreWindow()
    oProject = oDesktop.GetActiveProject()
    oDesign = oProject.GetActiveDesign()
    oEditor = oDesign.SetActiveEditor("Layout")
except Exception:
    oEditor = None

# Store config in user's home directory so it persists across sessions safely
CONFIG_FILE = os.path.join(os.path.expanduser('~'), '.aedt_spiral_config.json')

def polygon_spiral_points(n_sides, inner_radius, width, spacing, n_turns):
    step = width + spacing
    calc_pi = math.pi / float(n_sides)
    A = 2.0 * inner_radius * math.sin(calc_pi)
    B = (2.0 * step / float(n_sides)) * math.tan(calc_pi)
    
    start_x = -A / 2.0
    start_y = -inner_radius * math.cos(calc_pi)
    
    points = [(start_x, start_y)]
    x, y = start_x, start_y
    
    # Use math.ceil to cleanly handle floating point turns (like 4.4)
    total_segments = int(math.ceil(n_turns * n_sides))
    for k in range(total_segments):
        L_k = A + B * k
        angle = k * (2.0 * math.pi / n_sides)
        x += L_k * math.cos(angle)
        y += L_k * math.sin(angle)
        points.append((x, y))
        
    return points

def create_polygon_spiral(n_sides, layer, inner_radius, width, spacing, n_turns):
    pts = polygon_spiral_points(n_sides, inner_radius, width, spacing, n_turns)
    n = len(pts)

    coords = []
    for (x, y) in pts:
        coords += ["x:=", round(x, 6), "y:=", round(y, 6)]

    line_name = "PolySpiral_N{}_L{}_R{}".format(int(n_sides), float(n_turns), float(inner_radius)).replace('.', 'p')

    if oEditor:
        oEditor.CreateLine(
            [
                "NAME:Contents",
                "lineGeometry:=",
                [
                    "Name:=",       line_name,
                    "LayerName:=",  layer,
                    "lw:=",         "{:.6g}mm".format(width),
                    "endstyle:=",   0,
                    "StartCap:=",   0,
                    "n:=",          n,
                    "U:=",          "mm"
                ] + coords
            ]
        )
        print("Created '{}' with {} points on layer '{}'.".format(line_name, n, layer))
    else:
        print("AEDT Editor not found. Generated points only.")


class SpiralForm(Form):
    def __init__(self):
        self.Text = "Polygon Spiral Generator"
        self.Width = 330
        self.Height = 420
        
        # Window styling
        self.FormBorderStyle = FormBorderStyle.FixedDialog
        self.MaximizeBox = False
        self.MinimizeBox = False
        self.StartPosition = FormStartPosition.CenterScreen
        self.BackColor = Color.WhiteSmoke
        
        # Default font
        self.Font = Font("Segoe UI", 9)
        
        # Default config
        self.config = {
            "N_SIDES": 12,
            "LAYER": "SIwave Regions",
            "INNER_RADIUS": 5.0,
            "WIDTH": 0.5,
            "SPACING": 0.5,
            "TURNS": 4.4
        }
        
        # Load from JSON if exists
        if os.path.exists(CONFIG_FILE):
            try:
                with open(CONFIG_FILE, 'r') as f:
                    saved = json.load(f)
                    self.config.update(saved)
            except Exception:
                pass
                
        self.inputs = {}
        y_offset = 20
        
        # Setup form dynamically
        keys = ["N_SIDES", "LAYER", "INNER_RADIUS", "WIDTH", "SPACING", "TURNS"]
        
        # Get active layers
        available_layers = []
        if oEditor:
            try:
                available_layers = list(oEditor.GetAllLayerNames())
            except Exception:
                pass
        
        if not available_layers:
            available_layers = ["SIwave Regions"] # fallback
        
        for key in keys:
            lbl = Label()
            lbl.Text = key
            lbl.Location = Point(20, y_offset + 3)
            lbl.Width = 100
            lbl.Font = Font("Segoe UI", 8, FontStyle.Bold)
            self.Controls.Add(lbl)
            
            if key == "LAYER":
                box = ComboBox()
                box.DropDownStyle = ComboBoxStyle.DropDownList
                for l in available_layers:
                    box.Items.Add(l)
                if self.config.get(key) in available_layers:
                    box.SelectedItem = self.config[key]
                else:
                    box.SelectedIndex = 0
            else:
                box = TextBox()
                box.Text = str(self.config.get(key, ""))
                box.TextChanged += self.on_text_changed
                
            box.Location = Point(140, y_offset)
            box.Width = 140
            self.Controls.Add(box)
            self.inputs[key] = box
            
            y_offset += 35
            
        self.error_label = Label()
        self.error_label.ForeColor = Color.Crimson
        self.error_label.Location = Point(20, y_offset)
        self.error_label.Width = 260
        self.error_label.Height = 25
        self.Controls.Add(self.error_label)
        
        y_offset += 30
        
        # Create Button
        self.btn_ok = Button()
        self.btn_ok.Text = "Create Spiral"
        self.btn_ok.Font = Font("Segoe UI", 9, FontStyle.Bold)
        self.btn_ok.Size = Size(130, 42)
        self.btn_ok.Location = Point(20, y_offset)
        self.btn_ok.BackColor = Color.DodgerBlue
        self.btn_ok.ForeColor = Color.White
        self.btn_ok.FlatStyle = FlatStyle.Flat
        self.btn_ok.FlatAppearance.BorderSize = 0
        self.btn_ok.Click += self.on_ok
        self.Controls.Add(self.btn_ok)
        
        # Close Button
        self.btn_cancel = Button()
        self.btn_cancel.Text = "Close"
        self.btn_cancel.Font = Font("Segoe UI", 9)
        self.btn_cancel.Size = Size(130, 42)
        self.btn_cancel.Location = Point(160, y_offset)
        self.btn_cancel.BackColor = Color.LightGray
        self.btn_cancel.ForeColor = Color.Black
        self.btn_cancel.FlatStyle = FlatStyle.Flat
        self.btn_cancel.FlatAppearance.BorderSize = 0
        self.btn_cancel.Click += self.on_cancel
        self.Controls.Add(self.btn_cancel)
        
        # Run validation once to set initial state
        self.validate_inputs()
        
    def on_text_changed(self, sender, event):
        self.validate_inputs()

    def validate_inputs(self):
        all_valid = True
        error_msg = ""
        
        for key, box in self.inputs.items():
            if isinstance(box, ComboBox):
                continue
                
            box.BackColor = Color.White
            val_text = box.Text.strip()
            
            try:
                if key == "N_SIDES":
                    val = int(val_text)
                    if val < 3:
                        raise ValueError("must be integer >= 3")
                else:
                    val = float(val_text)
                    if val <= 0:
                        raise ValueError("must be > 0")
            except Exception as e:
                box.BackColor = Color.LightPink
                all_valid = False
                err_str = str(e)
                if "must be" not in err_str:
                    error_msg = "{} requires a valid number".format(key)
                else:
                    error_msg = "{} {}".format(key, err_str)
                    
        if all_valid:
            self.error_label.Text = ""
            self.btn_ok.Enabled = True
            self.btn_ok.BackColor = Color.DodgerBlue
        else:
            self.error_label.Text = error_msg
            self.btn_ok.Enabled = False
            self.btn_ok.BackColor = Color.LightGray

    def on_ok(self, sender, event):
        if not self.btn_ok.Enabled:
            return
            
        try:
            # Typecasting validation
            self.config["N_SIDES"] = int(self.inputs["N_SIDES"].Text)
            
            if isinstance(self.inputs["LAYER"], ComboBox):
                self.config["LAYER"] = str(self.inputs["LAYER"].SelectedItem)
            else:
                self.config["LAYER"] = str(self.inputs["LAYER"].Text)
                
            self.config["INNER_RADIUS"] = float(self.inputs["INNER_RADIUS"].Text)
            self.config["WIDTH"] = float(self.inputs["WIDTH"].Text)
            self.config["SPACING"] = float(self.inputs["SPACING"].Text)
            self.config["TURNS"] = float(self.inputs["TURNS"].Text)
            
            # Save configuration
            with open(CONFIG_FILE, 'w') as f:
                json.dump(self.config, f)
                
            # Execute Generation
            create_polygon_spiral(
                self.config["N_SIDES"],
                self.config["LAYER"],
                self.config["INNER_RADIUS"],
                self.config["WIDTH"],
                self.config["SPACING"],
                self.config["TURNS"]
            )
        except Exception as e:
            MessageBox.Show("Error processing input: " + str(e))
            
    def on_cancel(self, sender, event):
        self.Close()

def main():
    form = SpiralForm()
    Application.Run(form)

if __name__ == '__main__':
    main()
```
